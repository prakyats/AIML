import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.model_selection import train_test_split as tts
from sklearn.linear_model import LinearRegression as lr
from sklearn.metrics import mean_squared_error as mse , r2_score as r2s , mean_absolute_error as mae
from sklearn.preprocessing import StandardScaler as ss

df=pd.read_csv("economic_index.csv")
df

df.head()
df.describe()
df.info()
df.isnull().sum()
df.shape
sns.pairplot(df)

plt.scatter(df['interest_rate'], df['unemployment_rate'], color='b')
plt.xlabel('interest_rate')
plt.ylabel('unemployment_rate')
plt.show()

X=df.iloc[:, : -1]
y=df.iloc[:,-1]

X_train, X_test, y_train , y_test = tts( X , y , random_state =42 , test_size=0.25 )
sns.regplot(x=df['interest_rate'],y=df['index_price'])

sns.regplot(x=df['unemployment_rate'],y=df['index_price'])

scaler=ss()
X_train=scaler.fit_transform(X_train)
X_test=scaler.fit_transform(X_test)
X_train,X_test

reg=lr()
reg.fit(X_train, y_train)

y_pred=reg.predict(X_test)
y_pred

print("MSE :",mse(y_test, y_pred) )
print("MAE :",mae(y_test, y_pred) )
